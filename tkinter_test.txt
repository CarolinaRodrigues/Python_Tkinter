#Aula 23 -> EXERCÍCIO FAHRENHEIT PARA CELSIUS

#ESCOPO DO CODIGO
from tkinter import *

#grid master

#-----------------------------------------------------------
#Funções
#C = (r-32) * 5/9
def calcular():
    F = float(textbox1.get())
    C = (F - 32) * 5/9
    final.set( str(round (C,1)) + "graus celsius")
#------------------------------------------------------------

#GUI
root = Tk()
root.title("Aplicação")
#root.geometry("200x100")
root.iconbitmap("images/icon1.ico")

final = StringVar()

#-------------------------------------------------------------------------
#Criando os Widgets
label1 = Label(root, text = "Graus Fahrenheit:")
textbox1 = Entry(root)
button1 = Button(root,text = "Calcular",command = calcular)
label_resultado = Label(root, textvariable = final)
#-------------------------------------------------------------------------
#Layout Widgets
label1.grid()
textbox1.grid()
button1.grid()
label_resultado.grid()
#-------------------------------------------------------------------------
root.mainloop()
